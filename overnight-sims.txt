# ============================================================================
#  overnight-sims.txt — the unattended batch. Launch, sleep, read the digest.
#
#  LAUNCH (keeps the Mac awake for the whole run; key must be in .env):
#
#     RUNS=50  caffeinate -i bash overnight-sims.txt        # ~a night for 2 agents
#     RUNS=100 caffeinate -i bash overnight-sims.txt        # longer — see note
#
#  WHAT IT DOES: for each flagship agent that's landed on main, runs the full
#  robustness matrix (one fixed agent × all its fixtures) N times per arm,
#  governed-vs-ungoverned, VERBOSE (every run's prompt + tool trace + oracle
#  verdict + Pherix action), with 429-backoff + per-scenario fault isolation so
#  one bad run never kills the batch. Writes per-fixture + rollup JSON and a
#  full verbose log per agent, then a single DIGEST.txt with the headline and a
#  ⚠ flag on anything that needs you (escapes / regression alarms / a crashed agent).
#
#  WAKE UP: open  reports/overnight/<timestamp>/DIGEST.txt  — top to bottom is
#  "what needs doing". Full evidence sits beside it (per-agent .log + JSON).
#
#  AGENTS: runs whatever is on main — enterprise always; devops only once #21
#  lands (auto-skipped until then, so this is safe to launch tonight regardless).
#
#  COST/TIME: real Claude API, sequential + backoff. ~500 runs/agent at N=50,
#  ~1000 at N=100; budget several hours per agent and tens of $ — scales with N.
#  For a single overnight covering BOTH agents, N=50 is the safe choice.
# ============================================================================
set -uo pipefail
cd "$(git rev-parse --show-toplevel 2>/dev/null || echo .)"

RUNS="${RUNS:-50}"
export PYTHONUNBUFFERED=1   # flush logs line-by-line → partial log survives a kill
TS="$(date +%Y%m%d-%H%M%S)"
OUT="reports/overnight/$TS"
mkdir -p "$OUT"
DIGEST="$OUT/DIGEST.txt"

# --- key check (fail fast, before sleeping) ---------------------------------
{ [ -n "${ANTHROPIC_API_KEY:-}" ] || grep -q '^ANTHROPIC_API_KEY=sk' .env 2>/dev/null; } \
  || { echo "✗  no ANTHROPIC_API_KEY in env or .env — aborting"; exit 1; }

# --- which flagships are live on main? (skip what hasn't landed) ------------
AGENTS=()
[ -d examples/dogfood/sims/enterprise ] && AGENTS+=("enterprise")
[ -d examples/dogfood/sims/devops ]     && AGENTS+=("devops")

{
  echo "OVERNIGHT SIM BATCH — $TS"
  echo "runs/arm/fixture: $RUNS    agents: ${AGENTS[*]:-none}"
  echo "started: $(date)"
  echo "============================================================"
} | tee "$DIGEST"

[ ${#AGENTS[@]} -eq 0 ] && { echo "no agents available — nothing to run"; exit 1; }

# --- run each agent; isolate failures so one crash can't abort the batch ----
for a in "${AGENTS[@]}"; do
  log="$OUT/$a.log"
  echo ""; echo "▶  $a  (runs=$RUNS)  $(date +%H:%M:%S)"
  set +e
  python -m "examples.dogfood.sims.$a" --runs "$RUNS" -v --out "$OUT/$a/" > "$log" 2>&1
  rc=$?
  set -e
  echo "   $a finished rc=$rc  $(date +%H:%M:%S)  → $log"

  # --- per-agent digest entry ---
  {
    echo ""
    echo "──────── $a  (exit $rc) ────────"
    if [ $rc -ne 0 ]; then
      echo "⚠  AGENT CRASHED (exit $rc) — see $log (last lines):"
      tail -n 15 "$log"
    else
      # the rollup + any alarm tokens — the headline and the action items
      grep -iE "rollup|ungoverned|governed|caught|escaped|regression|alarm|false[_ ]positive" "$log" | tail -n 25 \
        || echo "(no rollup lines matched — check $log directly)"
      if grep -qiE "regression|alarm" "$log"; then echo "⚠  REGRESSION/ALARM present — investigate."; fi
      if grep -qiE "escaped *[1-9]|[1-9][0-9]* escaped" "$log"; then echo "⚠  ESCAPES present — a policy gap to name."; fi
    fi
  } >> "$DIGEST"
done

{
  echo ""
  echo "============================================================"
  echo "finished: $(date)"
  echo "evidence: $OUT/  (per-agent .log + per-fixture/rollup JSON)"
} | tee -a "$DIGEST"

echo ""; echo "✓  DONE — read $DIGEST"
