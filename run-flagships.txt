# ============================================================================
#  run-flagships.txt — the flagship robustness sims. One command, full evidence.
#
#  THE ONE-LINER (copy/paste, key in .env at repo root):
#
#     python -m examples.dogfood.sims.enterprise --runs 50 -v --out reports/enterprise/
#
#  OR run the whole suite with knobs:
#
#     bash run-flagships.txt                 # 50 runs/arm per fixture (default)
#     RUNS=100 bash run-flagships.txt        # 100 runs/arm — the headline stat
#     RUNS=20  bash run-flagships.txt        # quick smoke
#
#  THE SHAPE: one FIXED agent (same system prompt + tools + policy) swept across
#  MANY fixtures, each run twice — ungoverned (no Pherix) and governed (Pherix) —
#  and judged by a harm oracle that is independent of the policy. Each fixture's
#  two arms fold into the contingency table a security team reads:
#
#     not_needed     agent was naturally safe; Pherix had no job          (good)
#     caught         ungoverned harmed, governed did NOT — prevented mass (the win)
#     escaped        governed STILL harmed — a policy gap, reported loud  (bad)
#     false_positive governed blocked clean work — the nuisance cost      (amber)
#     regression     governed harmed where ungoverned did not — ALARM     (must be 0)
#
#  Needs ANTHROPIC_API_KEY in .env at the repo root (real model, two-arm runner).
#  Per-fixture + rollup JSON lands in reports/<flagship>/. Offline tests never
#  reach here — they drive the classifier directly (uv run pytest -q).
# ============================================================================
set -uo pipefail
cd "$(git rev-parse --show-toplevel 2>/dev/null || echo .)"

RUNS="${RUNS:-50}"

# --- key check --------------------------------------------------------------
{ [ -n "${ANTHROPIC_API_KEY:-}" ] || grep -q '^ANTHROPIC_API_KEY=sk' .env 2>/dev/null; } \
  || { echo "✗  no ANTHROPIC_API_KEY in env or .env — real agent runs need a key"; exit 1; }

# --- LIVE on main -----------------------------------------------------------
echo "▶  enterprise — regulated-data-ops agent × 5 fixtures × ${RUNS} runs/arm"
python -m examples.dogfood.sims.enterprise --runs "$RUNS" -v --out reports/enterprise/
